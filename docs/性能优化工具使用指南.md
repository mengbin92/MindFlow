## 性能监控工具

### 基础用法

```typescript
import { performanceMonitor } from '@mindflow/core';

// 标记性能点
performanceMonitor.mark('operation-start');

// 执行你的代码
doSomething();

// 结束标记
performanceMonitor.endMark('operation-start');

// 打印性能报告
performanceMonitor.logReport();
```

### 测量异步操作

```typescript
const result = await performanceMonitor.measureAsync(
  'async-operation',
  async () => {
    // 异步操作
    return await fetchData();
  }
);
```

### 测量同步操作

```typescript
const result = performanceMonitor.measure('sync-operation', () => {
  // 同步操作
  return processData();
});
```

### 获取性能指标

```typescript
const metrics = performanceMonitor.getMetrics();
console.log('Startup Time:', metrics.startupTime);
console.log('First Render:', metrics.firstRenderTime);
console.log('Memory Usage:', metrics.memoryUsage);
```

### 在 React 组件中使用

```typescript
import { useRenderPerformance } from '@mindflow/core';

function MyComponent() {
  useRenderPerformance('MyComponent');

  return <div>...</div>;
}
```

### 使用装饰器

```typescript
import { MeasurePerformance } from '@mindflow/core';

class EditorService {
  @MeasurePerformance
  processContent(content: string) {
    // 这个方法的执行时间会被自动测量
    return content.trim();
  }
}
```

## 虚拟滚动组件

### 基础用法

```typescript
import { VirtualList } from '@mindflow/web';

function App() {
  const items = Array.from({ length: 10000 }, (_, i) => ({
    id: i,
    name: `Item ${i}`,
  }));

  return (
    <VirtualList
      items={items}
      renderItem={(item, index) => (
        <div key={index}>{item.name}</div>
      )}
      itemHeight={50}
      height={600}
    />
  );
}
```

### 可变高度模式

```typescript
<VirtualList
  items={items}
  renderItem={(item, index) => (
    <div key={index}>{item.content}</div>
  )}
  itemHeight={50}  // 默认高度
  height={600}
  variableHeight={true}
  getItemHeight={(item, index) => {
    // 根据内容动态计算高度
    return calculateHeight(item);
  }}
/>
```

### 滚动到指定项

```typescript
const [scrollToIndex, setScrollToIndex] = useState(100);

<VirtualList
  items={items}
  renderItem={renderItem}
  itemHeight={50}
  height={600}
  scrollToIndex={scrollToIndex}
/>
```

### 虚拟文件树

```typescript
import { VirtualFileTree } from '@mindflow/web';

function App() {
  return (
    <VirtualFileTree
      height={600}
      className="my-file-tree"
    />
  );
}
```

## 内存优化工具

### 内存管理器

```typescript
import { memoryManager, ResourceType } from '@mindflow/core';

// 注册资源
const resourceId = memoryManager.registerResource(
  ResourceType.EventListener,
  window,
  () => window.removeEventListener('resize', handler)
);

// 注销单个资源
memoryManager.unregisterResource(resourceId);

// 清理所有资源
memoryManager.cleanup();

// 获取资源统计
const stats = memoryManager.getResourceStats();
console.log('Event Listeners:', stats[ResourceType.EventListener]);
```

### React Hook: useResourceManager

```typescript
import { useResourceManager, ResourceType } from '@mindflow/core';

function MyComponent() {
  const { registerResource, cleanup } = useResourceManager();

  useEffect(() => {
    const handler = () => console.log('resize');

    registerResource(
      ResourceType.EventListener,
      window,
      () => window.removeEventListener('resize', handler),
      { event: 'resize' }
    );

    return () => cleanup();
  }, []);

  return <div>...</div>;
}
```

### WeakCache（不阻止垃圾回收）

```typescript
import { WeakCache } from '@mindflow/core';

const cache = new WeakCache<object, string>();

const key = { id: 1 };
cache.set(key, 'cached value');

// 当 key 没有其他引用时，会自动被垃圾回收
const value = cache.get(key);
```

### LRU Cache（最近最少使用缓存）

```typescript
import { LRUCache } from '@mindflow/core';

const cache = new LRUCache<string, any>(100); // 最多缓存 100 项

cache.set('key1', { data: 'value1' });
const value = cache.get('key1');

console.log('Cache size:', cache.size());
cache.clear(); // 清空缓存
```

### 批处理更新

```typescript
import { BatchProcessor } from '@mindflow/core';

const processor = new BatchProcessor(
  async (batch: Item[]) => {
    // 批量处理项目
    await updateDatabase(batch);
  },
  10,    // 每批处理 10 项
  100    // 延迟 100ms
);

// 添加项目到队列
items.forEach(item => processor.add(item));

// 清空队列
processor.clear();
```

## 优化后的编辑器组件

### 使用优化后的编辑器

```typescript
import { OptimizedEditor } from '@mindflow/web';

function App() {
  return (
    <OptimizedEditor
      initialValue="# Hello World"
      docId="my-document"
      theme="light"
      autoSave={true}
      autoSaveDelay={2000}
      onChange={(value) => console.log('Content changed:', value)}
      onThemeChange={(theme) => console.log('Theme changed:', theme)}
      onSaveStateChange={(state) => console.log('Save state:', state)}
    />
  );
}
```

### 优化特性

1. **React.memo** - 避免不必要的重渲染
2. **useMemo** - 缓存计算结果
3. **useCallback** - 稳定的函数引用
4. **防抖更新** - 减少预览更新频率
5. **requestAnimationFrame** - 与浏览器刷新周期同步

## 最佳实践

### 1. 性能监控

```typescript
// ✅ 推荐：使用性能监控
performanceMonitor.mark('feature-start');
// 执行功能代码
performanceMonitor.endMark('feature-start');

// ❌ 避免：不监控性能
// 直接执行代码，无法发现问题
```

### 2. 大数据列表

```typescript
// ✅ 推荐：使用虚拟滚动
<VirtualList
  items={largeDataSet}
  renderItem={renderItem}
  itemHeight={50}
  height={600}
/>

// ❌ 避免：直接渲染所有项目
{largeDataSet.map(item => (
  <div key={item.id}>{item.name}</div>
))}
```

### 3. 内存管理

```typescript
// ✅ 推荐：使用内存管理器
useEffect(() => {
  const resourceId = registerResource(
    ResourceType.EventListener,
    window,
    cleanup
  );
  return () => unregisterResource(resourceId);
}, []);

// ❌ 避免：不清理资源
useEffect(() => {
  window.addEventListener('resize', handler);
  // 忘记清理
}, []);
```

### 4. 组件优化

```typescript
// ✅ 推荐：使用 React.memo
const MyComponent = React.memo<Props>(({ data }) => {
  return <div>{data.value}</div>;
});

// ❌ 避免：不优化的组件
const MyComponent = ({ data }) => {
  return <div>{data.value}</div>;
};
```

### 5. 缓存计算

```typescript
// ✅ 推荐：使用 useMemo
const expensiveValue = useMemo(() => {
  return heavyCalculation(data);
}, [data]);

// ❌ 避免：每次渲染都计算
const expensiveValue = heavyCalculation(data);
```

## 常见问题

### Q: 虚拟滚动导致滚动条不准确？

A: 检查 `itemHeight` 是否正确。对于可变高度的内容，使用 `variableHeight` 和 `getItemHeight`。

### Q: 内存持续增长？

A: 使用 `memoryManager.getResourceStats()` 检查是否有未清理的资源。确保在 useEffect 返回清理函数。

### Q: 性能监控影响性能？

A: 性能监控在生产环境默认禁用。通过环境变量控制：

```typescript
const monitor = new PerformanceMonitor(
  process.env.NODE_ENV === 'development'
);
```

### Q: React.memo 不起作用？

A: 检查 props 是否稳定。使用 `useMemo` 和 `useCallback` 确保对象和函数引用稳定。
